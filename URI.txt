/country
    ADD ADMIN добавление новой страны
    GET PUBLIC получение списка стран
/country/{id}
    PATCH ADMIN обновление страны
    DELETE ADMIN удаление страны
    GET PUBLIC получение страны со спсиком исполнителей страны
/performer
    ADD ADMIN добавление нового исполнителя
/performer?search=
    GET PUBLIC получение списка исполнителей (сортировка по имени и популярности)
/performer/{id}
    PATCH ADMIN обновление исполнителя
    DELETE ADMIN удаление исполнителя
    GET PUBLIC получение исполнителя
/genre
    ADD ADMIN добавление нового жанра
    GET PUBLIC получение списка жанров
/genre/{id}
    PATCH ADMIN обновление жанра
    DELETE ADMIN удаление жанра
    GET PUBLIC получение жанра со списком пластинок жанра
/record
    ADD ADMIN добавление новой пластинки
    GET PUBLIC получение списка пластинок (сортировка по названию, цене, исполнителю, году и популярности)
        фильтрация по названию и треклисту, списку жанров, год (от и до), цена (от и до), доступные или все
/record/{id}
    PATCH ADMIN обновление пластинки
    DELETE ADMIN удаление пластинки
    GET PUBLIC получение пластинки
